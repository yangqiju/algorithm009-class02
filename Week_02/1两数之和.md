
### 两数之和
  
    public int[] twoSum(int[] nums, int target) {
        HashMap<Integer,Integer> map = new HashMap<>();
        for(int i=0;i<nums.length;i++){
            Integer index2= map.get(target-nums[i]);
            if(index2!=null){
                return new int[]{index2,i};
            }
            map.put(nums[i],i);
        }
        return null;
    }
